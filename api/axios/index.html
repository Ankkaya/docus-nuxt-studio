无感刷新 token

无感刷新 token 即接口请求 token 过期后，利用 accessToken 刷新 token，然后用新 token 重新请求调用失败的接口

需要注意的点有两个

  1. 当刷新 token 的时候，禁止其他接口重复吊起刷新 token
  2. 当 token 刷新完成后，重新调用请求失败的接口，因此需要在 token 刷新完成前，把这些请求暂存起来
// request.js

// 请求队列
let requestList = []
// 是否正在刷新
let isRefreshToken = false

// 创建 axios 实例
const service = axios.create()

// 响应拦截器
service.interceptors.response.use(
  async (res) => {
    let code = res.data.code || 200
    if (code === 401) {
      // 如果没有在刷新 token，则进行刷新
      if (!isRefreshToken) {
        // 设置刷新状态，正在刷新
        isRefreshToken = true
        // 刷新 token
        const refreshToken = await refreshToken()
        // 保存 token 信息到本地
        setToken(refreshToken)
        // 执行已暂存的请求
        requestList.forEach((cb) => cb())
        // 置空暂存的请求队列
        requestList = []
        // 设置刷新状态为 false
        isRefreshToken = false
        // 重新发起当前请求
        return service(res.config)
      } else {
        // 如果正在刷新 token，暂存其他请求
        new Promise((resolve) => {
          requestList.push(() => {
            res.config.headers['Authorization'] = 'Bearer ' + getToken()
            resolve(service(res.config))
          })
        })
      }
    }
  }
)

Blob 响应处理

封装 http response 拦截器

// request.js

import axios from 'axios'

const service = axios.create({})

service.interceptors.response.use(
  (response) => {
     // 如果请求返回类型是 blob，直接返回结果
     if (response.config.responseType === 'blob') {
      if (response.status === 200) {
        return response
      } else {
        // 请求失败，直接提示错误信息
        return
      }
     }
  }
)

const manageDownloadResult = (result) => {
  // 如果返回类型是 application/json，代表请求失败，处理返回的 blob 流

    const reader = new FileReader(res.data)
    reader.readAsText(res.data)
    reader.onload = () => {
      const temp_res = JSON.parse(reader.result)
      // 处理返回结果的错误信息
    }
  } else {
    // 处理返回的 blob 流数据
    const blob = new Blob([res.data], { type: 'application/octet-stream;charset=UTF-8' })
    const contentDisposition = res.headers['content-disposition']
    // RegExp 中 \\. 
    // \. 转义 .
    // \\ 转义 \
    const pattern = new RegExp('filename=([^;]+\\.[^\\.;]+);*')
    const link = document.createElement('a')
    link.href = URL.createObjectURL(blob)
    link.download = decodeURIComponent(pattern.exec(contentDisposition)[1])
    link.click()
    // 释放之前创建的对象
    window.URL.revokeObjectURL(link.href)
  }
}
